# Functie om letters van het geraden woord te controleren en kleuren toe te wijzen
FUNCTIE controleer_letters(te_raden_woord, invoer, geraden_letters, geraden_letters_zonder_kleur)
    VOOR elke letter in het geraden woord
        ALS de letter correct en op de juiste positie is
            Kleur de letter groen
        ANDERS ALS de letter in het woord voorkomt maar op een verkeerde positie
            Kleur de letter geel
        ANDERS
            Zet de letter als een underscore "_"
EINDE FUNCTIE

# Functie voor het raden van het woord
FUNCTIE raad_woord(te_raden_woord, geraden_letters)
    INITIALISEER aantal pogingen op 0
    TERWIJL aantal pogingen < 5
        Vraag de speler om het woord in te voeren
        ALS de lengte van de invoer ongelijk is aan de lengte van het te raden woord
            Toon foutmelding voor onjuiste lengte
            GA verder naar de volgende poging
        ANDERS
            Controleer de letters met controleer_letters functie
            TOON de voortgang van de letters (groen voor correct, geel voor incorrect)
            ALS het geraden woord correct is
                Toon "Gefeliciteerd, je hebt het woord geraden!"
                Geef TRUE terug
            ANDERS
                Toon "Niet correct, probeer het opnieuw"
                Verhoog het aantal pogingen
    NA 5 pogingen
        Toon "Helaas, je hebt het woord niet geraden"
        Geef FALSE terug
EINDE FUNCTIE
